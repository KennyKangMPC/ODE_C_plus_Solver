60 void OdeBase::ode_jac (double *solin, double **Jout) {
61 
62  //check space was allocated for Jacobian
63  if ( !need_jac_ ) {
64  printf("the Jacobian matrix wasn't allocated for whichever solver was chosen (need_jac_ == false)");
65  exit(EXIT_FAILURE);
66  }
67 
68  unsigned long i,j;
69  double delsol;
70 
71  //get derivatives at current position
72  ode_fun_(solin, f_);
73  //perturb and calculate finite differences
74  for (i=0; i<neq_; i++) {
75  //perturb
76  delsol = ode_max2(absjacdel_, solin[i]*reljacdel_);
77  solin[i] += delsol;
78  //evaluate function
79  ode_fun_(solin, g_);
80  //compute approximate derivatives
81  for (j=0; j<neq_; j++) Jout[j][i] = (g_[j] - f_[j])/delsol;
82  //unperturb
83  solin[i] -= delsol;
84  }
85 }
