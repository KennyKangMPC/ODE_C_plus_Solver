199 bool OdeAdaptive::step_adaptive_ (double dt, bool extra) {
200 
201  //store the current solution in case the step is rejected
202  memcpy(solprev_, sol_, neq_*sizeof(double));
203  //store the time step for access in derived classes
204  dt_ = dt;
205  //call the basic stepper
206  step_(dt);
207  //count the step, even if it's subsequently rejected
208  nstep_++;
209  //execute adaptive calculations
210  adapt(abstol_, reltol_);
211  //see if the step should be rejected
212  if ( is_rejected() ) {
213  //reject the step by putting solprev back into sol
214  memcpy(sol_, solprev_, neq_*sizeof(double));
215  //count the rejection
216  nrej_++;
217  //return failure
218  return(false);
219  } else {
220  //increment the time
221  t_ += dt;
222  //check for nans and infs
223  if (nstep_ % icheck_ == 0) check_sol_integrity();
224  //do any extra stuff
225  if (extra) after_step(t_);
226  //return success
227  return(true);
228  }
229 }
